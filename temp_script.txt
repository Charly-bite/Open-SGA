              async directPrintLabels() {

              return;
          }

                      item_ids: this.selectedItems,
                      template_id: this.selectedTemplate !== 'default' ? this.selectedTemplate : null,
                      rotation: rotation
                  })
              });

                  this.directPrinting = false;

              const data = await res.json();

                  this.directPrinting = false;
                  return;

              // 2. Send images to the local Print Agent

              async searchProducts() {
          // Sanitize special characters to prevent search/render loops
          const sanitized = this.searchQuery.replace(/[\\\/\%\*\+\$]/g, '');
          if (sanitized !== this.searchQuery) {
              this.searchQuery = sanitized;
              return; // changing the model will trigger another debounced input event
          }
  
          const currentQuery = this.searchQuery;
          if (currentQuery.length < 1) {
              this.searchResults = [];
              return;
          }
          try {
              const res = await fetch(`/products/search?q=${encodeURIComponent(currentQuery)}&limit=50`);
              if (!res.ok) {
                  throw new Error(`Error ${res.status}: ${res.statusText}`);
              }
              const data = await res.json();
              if (this.searchQuery === currentQuery) {
                  this.searchResults = data.products || [];
                  this.showResults = true;
              }
          } catch (e) {
              console.error('Error searching:', e);
              this.showToast('Error de conexión al buscar productos', 'error');
          }
      },

              async duplicateItem(item) {
          try {
              const res = await fetch('/labels/queue/add', {
                  method: 'POST',
                  headers: { 'Content-Type': 'application/json' },
                  body: JSON.stringify({
                      code: item.code,
                      quantity: item.quantity,
                      peso_tara: item.peso_tara,
                      batch_number: item.batch_number,
                      batch_date: item.batch_date,
                      reinspection_date: item.reinspection_date,

                  })
              });

                  // Insert the duplicate right after the original item
                  const idx = this.queue.findIndex(i => i.id === item.id);

              } else {

              async loadFromSAP() {
          if (!this.sapOrderNumber || this.loadingSAP) return;
  
          // Prevent duplicate load via scanner ENTER bug
          if (this.sapOrder && this.sapOrder.DocNum == this.sapOrderNumber) {
              if (document.activeElement) document.activeElement.blur();
              return;
          }
  
          // Check if queue has items and ask for confirmation to clear
          if (this.queue.length > 0) {
              const confirmClear = confirm(
                  `La cola actual tiene ${this.queue.length} etiqueta(s).\n\n` +
                  `¿Deseas limpiar la cola antes de cargar el pedido SAP #${this.sapOrderNumber}?`
              );
              if (confirmClear) {
                  try {
                      await fetch('/labels/queue/clear', { method: 'POST' });
                      this.queue = [];
                      this.queueSelectedItems = [];
                  } catch (e) {
                      console.error('Error clearing queue:', e);
                  }
              } else {
                  // User wants to keep existing items and append new ones
                  const confirmAppend = confirm(
                      '¿Agregar los productos del pedido SAP a la cola existente?'
                  );
                  if (!confirmAppend) {
                      this.loadingSAP = false;
                      return; // User cancelled
                  }
              }
          }
  
          this.loadingSAP = true;
          try {
              const res = await fetch('/labels/sap/load', {
                  method: 'POST',
                  headers: { 'Content-Type': 'application/json' },
                  body: JSON.stringify({ order_number: this.sapOrderNumber })
              });
              const data = await res.json();
              if (data.success) {
                  this.sapOrder = data.order;
                  if (document.activeElement) document.activeElement.blur(); // Remove focus to prevent barcode scanner bug
                  // Add items to queue with better error handling
                  let addedCount = 0;
                  let skippedCount = 0;
                  const errors = [];
  
                  for (const item of data.order.items || []) {
                      const itemCode = item.ItemCode;
                      if (!itemCode) {
                          console.warn('Skipping item without ItemCode:', item);
                          skippedCount++;
                          continue;
                      }
                      try {
                          const addRes = await fetch('/labels/queue/add', {
                              method: 'POST',
                              headers: { 'Content-Type': 'application/json' },
                              body: JSON.stringify({
                                  code: itemCode,
                                  quantity: item.Quantity || 1,
                                  peso_tara: item.PesoTara || 0,
                                  batch_number: item.Lote || '',
                                  batch_date: item.FechaLote || '',
                                  reinspection_date: item.FechaVencimiento || '',
                                  warehouse: item.Warehouse || ''
                              })
                          });
                          const addData = await addRes.json();
                          if (addData.success) {
                              this.queue.push(addData.item);
                              this.queueSelectedItems.push(addData.item.id);
                              addedCount++;
                          } else {
                              console.warn(`Error adding ${itemCode}:`, addData.error);
                              errors.push(`${itemCode}: ${addData.error}`);
                          }
                      } catch (e) {
                          console.error(`Error adding ${itemCode}:`, e);
                          errors.push(`${itemCode}: Error de conexión`);
                      }
                  }
  
                  // Show summary with better formatting
                  if (addedCount > 0) {
                      let message = `Pedido SAP #${this.sapOrderNumber} cargado. ${addedCount} producto(s) agregado(s).`;
                      if (errors.length > 0) {
                          message += ` ${errors.length} con error.`;
                      }
                      this.$dispatch('notify', { msg: message, type: errors.length > 0 ? 'warning' : 'success' });
  
                      // Optional: detailed alert for errors if needed still, but toast is better.
                      if (errors.length > 0) {
                          let errMessage = `⚠️ ${errors.length} producto(s) no encontrado(s) en la base de datos:\n`;
                          errMessage += errors.slice(0, 5).join('\n');
                          if (errors.length > 5) {
                              errMessage += `\n... y ${errors.length - 5} más`;
                          }
                          setTimeout(() => alert(errMessage), 500);
                      }
                  } else if (errors.length > 0) {
                      this.$dispatch('notify', { msg: `No se pudo agregar ningún producto. ${errors.length} con error.`, type: 'error' });
                      let errMessage = `❌ No se pudo agregar ningún producto del pedido.\n\nProductos no encontrados:\n${errors.slice(0, 5).join('\n')}${errors.length > 5 ? '\n...' : ''}`;
                      setTimeout(() => alert(errMessage), 500);
                  } else {
                      this.$dispatch('notify', { msg: 'El pedido no contiene productos válidos.', type: 'warning' });
                  }
              } else {
                  this.$dispatch('notify', { msg: data.error || 'Error loading SAP order', type: 'error' });
              }
          } catch (e) {
              console.error('Error loading from SAP:', e);
              this.$dispatch('notify', { msg: 'Error de conexión con SAP: ' + e.message, type: 'error' });
          }
          this.loadingSAP = false;
      },
